Excel 파일로 내역 가져오기

현대카드 등 금융사에서 다운로드한 .xls (Excel 97-2003) 형식의 거래 내역 파일을 업로드하면 자동으로 내역을 분석하고 가계부에 추가해줍니다.

packages = ["pandas", "xlrd"] [terminal] selector = "#pyscript-status-container" autoclose = true import pandas as pd import io import js from pyodide.ffi import create_proxy import asyncio js_add_transaction = js.budgetApp.core.addTransaction async def process_excel_file(event): status_container = js.document.getElementById("pyscript-status-container") upload_label = js.document.getElementById("excel-upload-label") file_list = event.target.files if not file_list: return file = file_list.item(0) upload_label.innerText = file.name status_container.innerHTML = """

파일을 읽고 분석하는 중입니다...

""" await asyncio.sleep(0) try: file_data = await file.arrayBuffer() df = pd.read_excel(io.BytesIO(file_data), engine='xlrd') # --- ✨ 스크린샷을 보고 실제 엑셀 파일의 컬럼명에 맞게 수정했습니다. ✨ --- DATE_COLUMN = '이용일자' # '이용일'에서 '이용일자'로 변경 AMOUNT_COLUMN = '승인금액' # '이용금액'에서 '승인금액'으로 변경 TEXT_COLUMN = '가맹점명' # 이 컬럼명은 그대로 유지 # ----------------------------------------------------------- required_cols = [DATE_COLUMN, AMOUNT_COLUMN, TEXT_COLUMN] if not all(col in df.columns for col in required_cols): missing = [col for col in required_cols if col not in df.columns] raise ValueError(f"필수 컬럼({', '.join(missing)})이 파일에 없습니다.") df[DATE_COLUMN] = pd.to_datetime(df[DATE_COLUMN], errors='coerce').dt.strftime('%Y-%m-%d') df[AMOUNT_COLUMN] = pd.to_numeric(df[AMOUNT_COLUMN].astype(str).str.replace(r'[^0-9]', '', regex=True)).fillna(0).astype(int) df_filtered = df[df[AMOUNT_COLUMN] > 0].copy() expense_categories = js.budgetApp.config.categories.expense.to_py() def find_category_id(text): if not isinstance(text, str): return 'other' text_lower = text.lower() for category in expense_categories: for keyword in category['keywords']: if keyword.lower() in text_lower: return category['id'] return 'other' df_filtered['category'] = df_filtered[TEXT_COLUMN].apply(find_category_id) added_count = 0 total_count = len(df_filtered) transactions_to_add = df_filtered[[DATE_COLUMN, AMOUNT_COLUMN, TEXT_COLUMN, 'category']].to_dict('records') for record in transactions_to_add: new_tx = { "id": f"tx{js.Date.now() + js.Math.random()}", "type": 'expense', "amount": record[AMOUNT_COLUMN], "date": record[DATE_COLUMN], "text": record[TEXT_COLUMN], "category": record['category'] } is_duplicate = js.budgetApp.state.transactions.some(create_proxy( lambda t: t.date == new_tx['date'] and t.amount == new_tx['amount'] and t.text == new_tx['text'] )) if not is_duplicate: js_add_transaction(new_tx) added_count += 1 status_container.innerHTML = f"""

분석 완료!

총 {total_count}개의 내역 중 {added_count}개의 새로운 내역을 가계부에 추가했습니다.

(중복된 내역은 제외되었습니다)

""" await asyncio.sleep(2) js.budgetApp.ui.showSection('transactions') except Exception as e: status_container.innerHTML = f"""

오류 발생

{e}

파일 형식(.xls)이나 내부 컬럼명을 확인해주세요.

""" finally: event.target.value = "" upload_label.innerText = "다른 파일 선택" def setup(): file_input = js.document.getElementById("excel-file-upload") file_input.addEventListener("change", create_proxy(process_excel_file)) print("PyScript 핸들러가 설정되었습니다.") setup()
